# title: 나도코딩의 C언어 입문(3장 피라미드)


#include <stdio.h>
int main() {

	int floor, t=0;
	printf("몇 층으로 쌓겠습니까?(5~100) ");
	scanf_s("%d", &floor);

	for (int i = 0; i < 2*floor; i+=2)
	{
		for (int k = t; k<floor-1; ++k)
			printf(" ");
		for (int j = 0; j <= i; ++j)
			printf("*");
		printf("\n");
		t++;
	}
}
